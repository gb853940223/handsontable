左右移动选择+模糊搜索

左侧搜索使用的是请求接口搜索，右侧搜索使用的是本地前端搜索。如果基础样式无法以满足需求，可以自定义样式，将基础样式覆盖。

              
              
                <div class="switchselect"></div>
              
            
              
              
                var dataL = [];
                var dataR = [];
                for(var i = 0; i < 10; i ++){
                  var objTemp = {
                    name: 'nameL' + i,
                    id: i,                  
                  };
                  dataL.push(objTemp);
                }
                for(var k = 0; k < 5; k ++){
                  var objTemp1 = {
                    name: 'nameR' + (k+10),
                    id: k+10
                  };
                  dataR.push(objTemp1);
                }
                $('.switchselect1').switchselect({
                  dataL: dataL,
                  dataR: dataR,
                  searchUrl: '/aaa'
                });
